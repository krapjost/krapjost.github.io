package kata import “strings”

// My Solution func Accum(s string) string { uS := strings.ToUpper(s); var res string; for i := 0; i < len(uS); i++ { res += string(uS[i]) for k := 1; k <= i; k++ { res += string(strings.ToLower(string(uS[i]))) } if i == len(uS)-1 { break } res += “-” } return res }

// Best Practice func Accum2(s string) string { parts := make([]string, len(s))

for i := 0; i < len(s); i++ {
  parts[i] = strings.ToUpper(string(s[i])) + strings.Repeat(strings.ToLower(string(s[i])), i)
}

return strings.Join(parts, "-")

}

function likes1(names) { if (names.length === 0) { return ‘no one likes this’ } if (names.length === 1) { return ${names[0]} likes this } if (names.length === 2) { return ${names[0]} and ${names[1]} like this } if (names.length === 3) { return ${names[0]}, ${names[1]} and ${names[2]} like this } if (names.length >= 4) { return ${names[0]}, ${names[1]} and ${names.length -2} others like this } }

function likes2(names) { return { 0: no one likes this, 1: ${names[0]} likes this, 2: ${names[0]} and ${names[1]} like this, 3: ${names[0]}, ${names[1]} and ${names[2]} like this, 4: ${names[0]}, ${names[1]} and ${names.length-2} others like this, }[Math.min(names.length, 4)] }

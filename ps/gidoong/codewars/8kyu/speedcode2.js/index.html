/* Given two integer arrays a, b, both of length >= 1, create a program that returns true if the sum of the squares of each element in a is strictly greater than the sum of the cubes of each element in b. */

// 다른 사람의 해결법 const sumPwrs = (a, p) => a.reduce( (s, n) => s + n ** p, 0); const arrayMadness1 = (a, b) => sumPwrs(a, 2) > sumPwrs(b, 3);

// 나의 해결법 function arrayMadness2(a, b) { const sumSquareA = a.reduce((acc,cur)=>acc+cur2,0) const sumCubeB = b.reduce((acc,cur)=>acc+cur3,0) return sumSquareA>sumCubeB }
